<script>
const player = document.getElementById("player");
const obstacle = document.getElementById("obstacle");
const scoreText = document.getElementById("score");

let jumping = false;
let score = 0;
let obstacleX = 800;

function jump(){
    if(jumping) return;

    jumping = true;
    let height = 0;

    let up = setInterval(() => {
        height += 5;
        player.style.bottom = height + "px";

        if(height >= 120){
            clearInterval(up);

            let down = setInterval(() => {
                height -= 5;
                player.style.bottom = height + "px";

                if(height <= 0){
                    clearInterval(down);
                    jumping = false;
                }
            },15);
        }
    },15);
}

document.addEventListener("keydown", e=>{
    if(e.code==="Space"){
        jump();
    }
});

function gameLoop(){

    obstacleX -= 5;
    obstacle.style.left = obstacleX + "px";

    if(obstacleX < -25){
        obstacleX = 800;
        score++;
        scoreText.textContent = score;
    }

    const playerBottom =
        parseInt(player.style.bottom) || 0;

    if(
        obstacleX > 40 &&
        obstacleX < 90 &&
        playerBottom < 60
    ){
        alert("Game Over! Điểm: " + score);
        location.reload();
    }

    requestAnimationFrame(gameLoop);
}

gameLoop();
</script>
